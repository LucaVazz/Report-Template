%% CUSTOM ELEMENTS
\newcommand{\chapteronsamepage}[1]{{\let\clearpage\relax \vspace{123pt} \chapter{#1}}}

\newenvironment{absolutelynopagebreak}
  {\pagebreak\par\nobreak\vfil\penalty0\vfilneg
   \vtop\bgroup}
  {\par\xdef\tpd{\the\prevdepth}\egroup
   \prevdepth=\tpd}

\newcommand{\incompletemarker}[1]{
    \begin{quote}
        \colorbox{SupportingDeepOrange}{ \textcolor{white}{\MineSign \vspace{0.01cm} INCOMPLETE {\scriptsize #1}} }
    \end{quote} 
}

\providecommand{\tightlist}{
    \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}
}

\newcolumntype{$}{>{\global\let\currentrowstyle\relax}}
\newcolumntype{^}{>{\currentrowstyle}}
\newcommand{\rowstyle}[1]{\gdef\currentrowstyle{#1}%
  #1\ignorespaces
}
\newenvironment{stripedtable}[3]
	{
		\rowcolors{2}{#2!7}{white}
		\begin{tabular}{$#1}
			\arrayrulecolor{#2}
			\toprule
			\rowstyle{\bfseries}#3\\\midrule
    }
	{
			\bottomrule
		\end{tabular}
	}
